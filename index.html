function parsePiutangExcelData(rawRows) { if (!rawRows || rawRows.length === 0) return []; // Deteksi kolom otomatis berdasarkan nama header yang mirip let keys = Object.keys(rawRows[0]); let find = (syns) => keys.find(k => syns.some(s => k.toLowerCase().includes(s))) || keys[0]; let kCust = find(['cust', 'nama', 'toko', 'pelanggan']); let kTgl = find(['tanggal', 'tgl']); let kNom = find(['nominal', 'total', 'jumlah']); let kBay = find(['bayar', 'terbayar', 'pembayaran']); let kStat = find(['status', 'ket']); return rawRows.map(row => ({ customer: row[kCust] || '-', tanggal: row[kTgl] || '-', nominal: parseFloat(String(row[kNom]).replace(/[^0-9]/g, '')) || 0, terbayar: parseFloat(String(row[kBay]).replace(/[^0-9]/g, '')) || 0, status: row[kStat] || 'Belum Lunas' })); } function uploadPiutangExcel() { let file = document.getElementById('piutangFile').files[0]; if(!file) return; let reader = new FileReader(); reader.onload = function(e) { let workbook = XLSX.read(e.target.result, {type: 'binary'}); let data = XLSX.utils.sheet_to_json(workbook.Sheets[workbook.SheetNames[0]]); let processed = parsePiutangExcelData(data); localStorage.setItem('piutang_data', JSON.stringify(processed)); dataPiutang = processed; showNotification('Piutang berhasil diperbarui!', 'success'); resetPiutangUploadForm(); hitungDanTampilPiutang(); renderPiutangTable(); }; reader.readAsBinaryString(file); } function hitungDanTampilPiutang() { let totalNominal = dataPiutang.reduce((sum, item) => sum + item.nominal, 0); let totalTerbayar = dataPiutang.reduce((sum, item) => sum + item.terbayar, 0); document.getElementById('lblTotalPiutang').innerText = 'Rp ' + totalNominal.toLocaleString('id-ID'); document.getElementById('lblTotalTerbayar').innerText = 'Rp ' + totalTerbayar.toLocaleString('id-ID'); document.getElementById('lblSisaPiutang').innerText = 'Rp ' + (totalNominal - totalTerbayar).toLocaleString('id-ID'); } function renderPiutangTable() { let statusFilter = document.getElementById('filterStatusPiutang').value; let filtered = statusFilter ? dataPiutang.filter(i => i.status === statusFilter) : dataPiutang; let html = ``; filtered.forEach((r, idx) => { html += ``; }); html += `
CustomerTanggalNominalTerbayarStatus
${r.customer} ${r.tanggal} Rp ${r.nominal.toLocaleString('id-ID')} Rp ${r.terbayar.toLocaleString('id-ID')} ${r.status}
`; document.getElementById('piutangTableScroll').innerHTML = html; } function resetPiutangFilter() { document.getElementById('filterStatusPiutang').value = ''; renderPiutangTable(); }
